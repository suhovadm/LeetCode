# Remove Linked List Elements (Easy).

# Дано: голова связного списка и целое число val. Нужно удалить все узлы связного списка,
# у которых Node.val == val, и вернуть новую голову списка.

# Пример 1:
# вход: head = [1, 2, 6, 3, 4, 5, 6], val = 6
# выход: [1, 2, 3, 4, 5]

# Пример 2:
# вход: head = [], val = 1
# выход: []

# Пример 3:
# вход: head = [7, 7, 7, 7], val = 7
# выход: []

# Ограничения:
# количество узлов в списке находится в диапазоне [0, 10⁴]
# 1 ≤ Node.val ≤ 50
# 0 ≤ val ≤ 50

# Чего мы, собственно, пытаемся добиться ?
# Мы проходимся по списку и удаляем все узлы, у которых значение равно val, не создавая
# при этом новый список, а просто "перепрыгивая" через ненужные элементы с помощью ссылок.
# То есть, наша задача: сохранить порядок списка, но выкинуть все узлы с нужным значением.
