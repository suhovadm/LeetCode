# Excel Sheet Column Number

# Дана строка columnTitle, которая представляет название столбца в таблице Excel.
# Необходимо вернуть соответствующий номер этого столбца.

# Примеры:
# A - 1
# B - 2
# C - 3
# ...
# Z - 26
# AA - 27
# AB - 28
# ...

# Пример 1:
# Ввод: columnTitle = "A"
# Вывод: 1

# Пример 2:
# Ввод: columnTitle = "AB"
# Вывод: 28

# Пример 3:
# Ввод: columnTitle = "ZY"
# Вывод: 701
