# Excel Sheet Column Title

# Дано целое число columnNumber (номер столбца).
# Верните соответствующее ему название столбца, как в таблице Excel.

# Например:
# A - 1
# B - 2
# C - 3

# Z - 26
# AA - 27
# AB - 28

# Пример 1:
# вход: columnNumber = 1
# выход: "A"

# Пример 2:
# вход: columnNumber = 28
# выход: "AB"

# Пример 3:
# выход: columnNumber = 701
# выход: "ZY"

# Ограничения:
# 1 ≤ columnNumber ≤ 2³¹ − 1
