# Linked List Cycle (Easy, 141).

# Дан head - указатель на голову связного списка.
# Нужно определить, содержит ли связный список цикл.

# Цикл в связном списке есть, если существует такой узел, к которому можно вернуться,
# непрерывно переходя по указателям next. Внутри используется переменная pos, которая
# обозначает индекс узла, на который указывает next последнего элемента (хвоста).
# Обратите внимание, что pos не передаётся в функцию.

# Верните true, если в списке есть цикл, иначе верните false.

# Цикл создаётся тем, что последний узел (хвост) начинает указывать
# на то какой-то предыдущий узел по индексу (pos).

# Пример 1:
# вход: head = [3,2,0,-4], pos = 1
# выход: true
# Объяснение:
# в списке есть цикл: хвост указывает на узел с индексом 1 (нумерация с 0).

# Пример 2:
# вход: head = [1,2], pos = 0
# выход: true
# Объяснение:
# хвост указывает на узел с индексом 0.

# Пример 3:
# вход: head = [1], pos = -1
# выход: false
# Объяснение: цикла нет.

# Ограничения:
# Количество узлов в списке: от 0 до 10⁴
# Значение Node.val: от -10⁵ до 10⁵
# pos равен -1 или корректному индексу узла в списке

# pos - это сокращённо от position, т.е. место узла в списке.
# если pos != -1 - всегда есть цикл.
