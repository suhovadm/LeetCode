# Combination Sum

# Дан массив различных целых чисел candidates и целевое целое число target.
# Верните список всех уникальных комбинаций элементов из candidates, сумма
# которых равна target. Комбинации можно возвращать в любом порядке.

# Одно и то же число из candidates может выбираться неограниченное количество раз.
# Две комбинации считаются уникальными, если частота (количество вхождений) хотя
# бы одного из выбранных чисел в них различается.

# Тестовые случаи сгенерированны таким образом, что количество уникальных комбинаций,
# дающих в сумме target, для заданных входных данных меньше 150.

# Пример 1:
# Входные данные: candidates = [2,3,6,7], target = 7
# Выходные данные: [[2,2,3],[7]]
# Пояснение:
# 2 и 3 являются числами-кандидатами, и 2 + 2 + 3 = 7. Обратите внимание, что 2 можно
# использовать многократно.
# 7 является числом-кандидатом, и 7 = 7.
# Это единственно возможные две комбинации.

# Пример 2:
# Входные данные: candidates = [2,3,5], target = 8
# Выходные данные: [[2,2,2,2],[2,3,3],[3,5]]

# Пример 3:
# Входные данные: candidates = [2], target =  1
# Выходные данные: []

# Ограничения:
# 1 <= candidates.length <= 30 (длина массива candidates от 1 до 30)
# 2 <= candidates[i] <= 40 (значения элементов candidates от 2 до 40)
# Все элементы в candidates уникальны (различны).
# 1 <= target <= 40 (целевое число от 1 до 40).
