# Count Submatrices With Equal Frequency of X and Y

# Дан двумерный массив символов grid, где grid[i][j] - это либо 'X', 'Y', либо '.'.
# Необходимо вернуть количество подматриц, которые:

# - содержат элемент grid[0][0],
# - имеют равное количество символов 'X' и 'Y',
# - содержат хотя бы один символ 'X'.

# --- Пример 1:
# вход:
# grid = [["X", "Y", "."],
#        ["Y", ".", "."]]
# выход: 3

# --- Пример 2:
# вход:
# grid = [["X", "X"],
#        ["X", "Y"]]
# выход: 0
# Пояснение: ни одна подматрица не содержит равное количество 'X' и 'Y'.

# --- Пример 3:
# вход:
# grid = [[".", "."],
#        [".", "."]
# выход: 0
# Пояснение: ни одна подматрица не содержит хотя бы один символ 'X'.

# Ограничения:
# 1 <= grid.length, grid[i].length <= 1000
# grid[i][j] - это 'X', 'Y' или '.'
